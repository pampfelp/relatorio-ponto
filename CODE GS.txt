function jsonOutput(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── Helpers de aba ──────────────────────────────────────────────────

function getSheet(name) {
  var ss    = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(name) || findSheetLoose(ss, name);
  if (!sheet) throw new Error('Aba "' + name + '" não encontrada na planilha.');
  return sheet;
}

function getSheetSafe(name) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  return ss.getSheetByName(name) || findSheetLoose(ss, name);
}

function findSheetLoose(ss, name) {
  var target = normalizeSheetName(name);
  var sheets = ss.getSheets();
  for (var i = 0; i < sheets.length; i++) {
    if (normalizeSheetName(sheets[i].getName()) === target) return sheets[i];
  }
  return null;
}

function normalizeSheetName(s) {
  return String(s || '')
    .trim().toLowerCase()
    .normalize('NFD').replace(/[\u0300-\u036f]/g, '')
    .replace(/[_\s]+/g, ' ').trim();
}

// ── Leitura de planilha ─────────────────────────────────────────────

function sheetToObjects(sheet) {
  var values = sheet.getDataRange().getValues();
  if (values.length === 0) return [];
  var headers = values[0].map(function (h) { return String(h).trim(); });
  var out = [];
  for (var i = 1; i < values.length; i++) {
    var row = values[i];
    if (row.every(function (c) { return c === '' || c === null; })) continue;
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      if (!headers[j]) continue;
      var val = row[j];
      if (isDateLike(val)) val = formatDateBR(val);
      obj[headers[j]] = val;
    }
    out.push(obj);
  }
  return out;
}

/**
 * Equivalente ao "cache em memória tipo Redis" nesse ecossistema: o Apps
 * Script tem um CacheService nativo, compartilhado entre TODAS as
 * execuções do script (não só a atual), com expiração automática — exatamente
 * o papel que um Redis cumpriria aqui. Usado só pra abas que são lidas o
 * tempo todo e mudam pouco (Vendedores, Serviços) — não faz sentido pra
 * abas que mudam a cada request (Agendamentos, Vendas, Funil).
 *
 * Sempre que uma dessas abas for escrita, chame invalidarCache(chave) logo
 * depois — senão quem ler nos próximos minutos vai ver dado desatualizado.
 */
var SG_CACHE_TTL_PADRAO = 300; // 5 minutos — equilíbrio entre "atualizado" e "rápido"

function getSheetObjectsComCache(sheetName, chave, ttlSegundos) {
  var cache = CacheService.getScriptCache();
  var cacheado = cache.get(chave);
  if (cacheado) {
    try { return JSON.parse(cacheado); } catch (e) { /* cache corrompido — ignora e recarrega da planilha */ }
  }
  var sheet = getSheetSafe(sheetName);
  var dados = sheet ? sheetToObjects(sheet) : [];
  if (normalizeSheetName(sheetName) === normalizeSheetName(CONFIG.ABA_VENDEDORES)) {
    dados = sanitizeVendedoresPublico(dados);
  }
  try {
    var json = JSON.stringify(dados);
    if (json.length < 95000) cache.put(chave, json, ttlSegundos || SG_CACHE_TTL_PADRAO); // CacheService tem limite de ~100KB por chave
  } catch (e) { /* não é crítico — só não guarda cache dessa vez, continua funcionando normal */ }
  return dados;
}

/**
 * Remove os campos sensíveis de autenticação (hash/salt de senha, token de
 * redefinição) antes de qualquer vendedor sair pro cliente. Toda leitura de
 * Vendedores por getSheetObjectsComCache passa por aqui — login/troca de
 * senha não usam esse caminho, então continuam funcionando normalmente.
 */
function sanitizeVendedoresPublico(lista) {
  var camposOcultos = { SenhaHash: 1, SenhaSalt: 1, ResetToken: 1, ResetTokenExpira: 1 };
  return (lista || []).map(function (v) {
    var out = {};
    Object.keys(v).forEach(function (k) { if (!camposOcultos[k]) out[k] = v[k]; });
    return out;
  });
}

function invalidarCache(chave) {
  try { CacheService.getScriptCache().remove(chave); } catch (e) { /* não é crítico */ }
}

function abonosSheetToObjects(sheet) {
  var values     = sheet.getDataRange().getValues();
  if (values.length === 0) return [];
  var headers    = values[0].map(function (h) { return String(h).trim(); });
  var horaFields = { Entrada: true, Almoco: true, RetornoAlmoco: true, Saida: true };
  var out = [];
  for (var i = 1; i < values.length; i++) {
    var row = values[i];
    if (row.every(function (c) { return c === '' || c === null; })) continue;
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      if (!headers[j]) continue;
      var val = row[j];
      if      (horaFields[headers[j]]) val = formatHoraSomente(val);
      else if (isDateLike(val))         val = formatDateBR(val);
      obj[headers[j]] = val;
    }
    out.push(obj);
  }
  return out;
}

function isDateLike(val) {
  return val instanceof Date ||
    (val && typeof val.getFullYear === 'function' && typeof val.getMonth === 'function');
}

function formatDateBR(d) {
  return String(d.getDate()).padStart(2,'0') + '/' +
    String(d.getMonth()+1).padStart(2,'0') + '/' + d.getFullYear() + ' ' +
    String(d.getHours()).padStart(2,'0') + ':' +
    String(d.getMinutes()).padStart(2,'0') + ':' +
    String(d.getSeconds()).padStart(2,'0');
}

function formatHoraSomente(val) {
  if (val === '' || val === null || val === undefined) return '';
  if (isDateLike(val)) {
    return String(val.getHours()).padStart(2,'0') + ':' + String(val.getMinutes()).padStart(2,'0');
  }
  var m = String(val).trim().match(/^(\d{1,2}):(\d{2})/);
  return m ? m[1].padStart(2,'0') + ':' + m[2] : '';
}

function normalizeDateKey(val) {
  if (isDateLike(val)) {
    return val.getFullYear() + '-' +
      String(val.getMonth()+1).padStart(2,'0') + '-' + String(val.getDate()).padStart(2,'0');
  }
  var str = String(val).trim();
  var iso = str.match(/^(\d{4})-(\d{2})-(\d{2})/);
  if (iso) return iso[1] + '-' + iso[2] + '-' + iso[3];
  var br  = str.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})/);
  if (br)  return br[3] + '-' + br[2].padStart(2,'0') + '-' + br[1].padStart(2,'0');
  return null;
}

var ABONOS_HEADERS = ['Funcionario','Data','Abonado','Entrada','Almoco','RetornoAlmoco','Saida'];

function getOrCreateAbonosSheet() {
  var ss    = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(CONFIG.ABA_ABONOS);
  if (!sheet) {
    sheet = ss.insertSheet(CONFIG.ABA_ABONOS);
    sheet.getRange(1,1,1,7).setValues([ABONOS_HEADERS]);
    sheet.setFrozenRows(1);
  } else {
    var first = String(sheet.getRange(1,1).getValue()||'').trim();
    if (first !== 'Funcionario') {
      sheet.insertRowBefore(1);
      sheet.getRange(1,1,1,7).setValues([ABONOS_HEADERS]);
      sheet.setFrozenRows(1);
    }
  }
  sheet.getRange('B2:B').setNumberFormat('@STRING@');
  sheet.getRange('D2:G').setNumberFormat('@STRING@');
  return sheet;
}

/**
 * Gera um ID curto tipo "a3f9c21b" (8 caracteres hex)
 */
/**
 * Garante que um valor monetário/numérico vindo do front-end (que sempre
 * manda no formato americano — ponto decimal, ex: "39.9" — porque vem de
 * <input type="number">) seja gravado na planilha como NÚMERO de verdade,
 * nunca como texto. Se ficar como texto, o front-end (que trata qualquer
 * texto como formato brasileiro, ponto = milhar) lê errado depois —
 * foi exatamente isso que causou o bug de "39,90" virar "3990".
 */
/**
 * Pega só os dígitos do telefone (ignora hífen, espaço, parênteses, +55…) e
 * devolve os últimos 8 — usado pra detectar cliente duplicado mesmo quando
 * o número foi digitado em formatos diferentes. Espelha exatamente a mesma
 * lógica do front-end (window.SGUtil.ultimos8DigitosTelefone).
 */
function paraUltimos8Digitos(tel) {
  var digitos = String(tel || '').replace(/\D/g, '');
  return digitos.length >= 8 ? digitos.slice(-8) : null;
}

function paraNumero(v) {
  if (v === undefined || v === null || v === '') return '';
  if (typeof v === 'number') return v;
  var n = parseFloat(String(v).replace(',', '.'));
  return isNaN(n) ? '' : n;
}

function generateShortId() {
  return Utilities.getUuid().replace(/-/g,'').substring(0,8);
}

// ── Actions: Ponto ─────────────────────────────────────────────────

function getData() {
  var vendedores  = getSheetObjectsComCache(CONFIG.ABA_VENDEDORES, 'sh_vendedores');
  var ponto       = sheetToObjects(getSheet(CONFIG.ABA_PONTO));
  var abonosSheet = getOrCreateAbonosSheet();
  var abonosRaw   = abonosSheetToObjects(abonosSheet);

  var overrides = {};
  abonosRaw.forEach(function (row) {
    if (!row.Funcionario || !row.Data) return;
    var dataKey = normalizeDateKey(row.Data);
    if (!dataKey) return;
    overrides[String(row.Funcionario) + '|' + dataKey] = {
      abonado: String(row.Abonado).toUpperCase() === 'TRUE' || row.Abonado === true,
      entrada: row.Entrada || '', almoco: row.Almoco  || '',
      retorno: row.RetornoAlmoco || '', saida: row.Saida || ''
    };
  });

  return { ok: true, vendedores: vendedores, ponto: ponto, overrides: overrides,
           atualizadoEm: new Date().toISOString() };
}

function setOverride(body) {
  var funcionario = body.funcionario, data = body.data;
  if (!funcionario || !data) return { ok: false, erro: 'funcionario e data são obrigatórios.' };

  var sheet  = getOrCreateAbonosSheet();
  var values = sheet.getDataRange().getValues();
  var rowIdx = -1;
  for (var i = 1; i < values.length; i++) {
    if (String(values[i][0]) === String(funcionario) &&
        normalizeDateKey(values[i][1]) === String(data)) {
      rowIdx = i + 1; break;
    }
  }

  var abonado = !!body.abonado, entrada = body.entrada||'', almoco = body.almoco||'',
      retorno = body.retorno||'', saida = body.saida||'';

  if (!abonado && !entrada && !almoco && !retorno && !saida) {
    if (rowIdx !== -1) sheet.deleteRow(rowIdx);
    return { ok: true, acao: 'removido' };
  }

  var rowValues = [funcionario, data, abonado, entrada, almoco, retorno, saida];
  if (rowIdx !== -1) {
    sheet.getRange(rowIdx,1,1,7).setValues([rowValues]);
    return { ok: true, acao: 'atualizado' };
  } else {
    sheet.appendRow(rowValues);
    return { ok: true, acao: 'criado' };
  }
}

/**
 * action: salvarVenda — body { idVenda?, idCliente, idServico, idVendedor, dataVenda ('YYYY-MM-DD'), valor }
 * Cria ou atualiza uma linha na aba Vendas.
 */
function actionSalvarVenda(body) {
  var idCliente = body.idCliente, idServico = body.idServico, idVendedor = body.idVendedor,
      dataVenda = body.dataVenda, valor = body.valor;
  if (!idCliente || !idServico || !idVendedor || !dataVenda || valor === undefined || valor === '') {
    return { ok: false, erro: 'Cliente, serviço, vendedor, data e valor são obrigatórios.' };
  }
  var sheet = getSheetSafe(CONFIG.ABA_VENDAS);
  if (!sheet) return { ok: false, erro: 'Aba Vendas não encontrada.' };
  var data = body.idVenda ? sheet.getDataRange().getValues() : [getHeaderRow(sheet)];
  var headers = data.length ? data[0].map(function (h) { return String(h).trim(); }) : ['IdVenda', 'IdCliente', 'IdServico', 'IdVendedor', 'DataVenda', 'Valor'];
  var colIdx = {}; headers.forEach(function (h, i) { colIdx[h] = i + 1; });

  var rowIndex = -1;
  if (body.idVenda && colIdx['IdVenda']) {
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][colIdx['IdVenda'] - 1]) === String(body.idVenda)) { rowIndex = i + 1; break; }
    }
  }

  var campos = {
    'IdCliente': idCliente, 'IdServico': idServico, 'IdVendedor': idVendedor,
    'DataVenda': parseDataISO(dataVenda), 'Valor': paraNumero(valor)
  };

  var novoId = body.idVenda;
  if (rowIndex === -1) {

/**
 * action: excluirVenda — body { idVenda }
 * Remove a linha da aba Vendas. Os custos vinculados em Custos_da_Venda não
 * são apagados automaticamente (ficam órfãos, sem efeito prático).
 */
function actionExcluirVenda(body) {
  var idVenda = body.idVenda;
  if (!idVenda) return { ok: false, erro: 'idVenda é obrigatório.' };
  var sheet = getSheetSafe(CONFIG.ABA_VENDAS);
  if (!sheet) return { ok: false, erro: 'Aba Vendas não encontrada.' };
  var data = sheet.getDataRange().getValues();
  var idCol = -1, vendedorCol = -1;
  for (var c = 0; c < data[0].length; c++) {
    var h = String(data[0][c]).trim();
    if (h === 'IdVenda') idCol = c;
    if (h === 'IdVendedor') vendedorCol = c;
  }
  if (idCol === -1) return { ok: false, erro: 'Coluna IdVenda não encontrada.' };
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][idCol]) === String(idVenda)) {
      var dono = vendedorCol !== -1 ? String(data[i][vendedorCol]) : null;

        return { ok: false, erro: 'Você só pode excluir as próprias vendas.' };
      }
      sheet.deleteRow(i + 1);
      return { ok: true };
    }
  }
  return { ok: false, erro: 'Venda não encontrada.' };
}

  };

  if (rowIndex === -1) {
    novoId = novoId || generateShortId(); // usa o ID que o cliente já gerou (otimista) — só gera aqui se não veio
    var lastCol = sheet.getLastColumn();

/**
 * action: salvarFunil — body { idOportunidade?, idCliente, idVendedor, idServico?,
 *   etapa, observacoes?, valorEstimado?, motivoPerda? }
 * Cria ou atualiza um lead do Funil. Se a etapa mudou (ou é lead novo), registra
 * a transição na aba FunilLog — é isso que alimenta o cálculo de "dias na
 * etapa" e a cor de SLA na tela, então precisa ficar consistente com a UI.
 */
function actionSalvarFunil(body) {
  var idCliente = body.idCliente, idVendedor = body.idVendedor, etapa = body.etapa;
  if (!idCliente || !idVendedor || !etapa) {
    return { ok: false, erro: 'Cliente, vendedor e etapa são obrigatórios.' };
  }

  var sheet = getSheetSafe(CONFIG.ABA_FUNIL);
  if (!sheet) return { ok: false, erro: 'Aba Funil não encontrada.' };
  var data = body.idOportunidade ? sheet.getDataRange().getValues() : [getHeaderRow(sheet)];
  var headers = data[0].map(function (h) { return String(h).trim(); });
  var colIdx = {}; headers.forEach(function (h, i) { colIdx[h] = i + 1; });

  var idCol = colIdx['IdOportunidade'];
  if (!idCol) return { ok: false, erro: 'Coluna IdOportunidade não encontrada na aba Funil.' };

  var rowIndex = -1, etapaAnterior = null;
  if (body.idOportunidade) {
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][idCol - 1]) === String(body.idOportunidade)) {
        rowIndex = i + 1;
        etapaAnterior = colIdx['Etapa'] ? String(data[i][colIdx['Etapa'] - 1] || '').trim() : null;
        break;
      }
    }
  }

  var agora = new Date();
  var campos = {
    'IdCliente': idCliente,
    'IdVendedor': idVendedor,
    'IdServico': body.idServico || '',
    'Etapa': etapa,
    'Observacoes': body.observacoes || '',
    'Valor Estimado': body.valorEstimado !== undefined ? body.valorEstimado : '',
    'Motivo Perda': body.motivoPerda || ''
  };

  var novoId = body.idOportunidade;
  if (rowIndex === -1) {
    novoId = novoId || generateShortId(); // usa o ID que o cliente já gerou (otimista) — só gera aqui se não veio
    var lastCol = sheet.getLastColumn();
    var novaLinha = new Array(lastCol).fill('');
    Object.keys(colIdx).forEach(function (h) {
      if (h === 'IdOportunidade') novaLinha[colIdx[h] - 1] = novoId;
      else if (h === 'Data Criacao') novaLinha[colIdx[h] - 1] = agora;
      else if (campos[h] !== undefined) novaLinha[colIdx[h] - 1] = campos[h];
    });
    sheet.getRange(sheet.getLastRow() + 1, 1, 1, novaLinha.length).setValues([novaLinha]);
  } else {
    Object.keys(campos).forEach(function (h) {
      if (colIdx[h]) sheet.getRange(rowIndex, colIdx[h]).setValue(campos[h]);
    });
  }

  if (etapaAnterior === null || etapaAnterior !== etapa) {
    registrarFunilLog(novoId, etapaAnterior || '', etapa, agora);
  }

  return { ok: true, idOportunidade: novoId };
}

/**
 * Lê só a LINHA DE CABEÇALHO de uma aba, sem trazer os dados inteiros —
 * muito mais rápido que getDataRange() em abas grandes (Clientes, Ponto,
 * FunilLog, etc.), usado sempre que só precisamos saber em que coluna cada
 * campo está, não os dados já existentes.
 */
function getHeaderRow(sheet) {
  var lastCol = sheet.getLastColumn();
  if (lastCol === 0) return [];
  return sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(function (h) { return String(h).trim(); });
}

function registrarFunilLog(idOportunidade, etapaAnterior, etapaNova, quando) {
  var logSheet = getSheetSafe(CONFIG.ABA_FUNIL_LOG);
  if (!logSheet) return;
  var headers = getHeaderRow(logSheet);
  var linha = headers.map(function (h) {
    if (h === 'IdOportunidade') return idOportunidade;
    if (h === 'EtapaAnterior') return etapaAnterior;
    if (h === 'EtapaNova') return etapaNova;
    if (h === 'Datahora' || h === 'DataHora') return quando;
    return '';
  });
  logSheet.getRange(logSheet.getLastRow() + 1, 1, 1, linha.length).setValues([linha]);
}

/**
 * action: excluirFunil — body { idOportunidade }
 * Remove a linha do lead da aba Funil. Não apaga o histórico em FunilLog
 * (fica como registro, não afeta nada já que o lead não existe mais na lista).
 */
function actionExcluirFunil(body) {
  var idOportunidade = body.idOportunidade;
  if (!idOportunidade) return { ok: false, erro: 'idOportunidade é obrigatório.' };
  var sheet = getSheetSafe(CONFIG.ABA_FUNIL);
  if (!sheet) return { ok: false, erro: 'Aba Funil não encontrada.' };
  var data = sheet.getDataRange().getValues();
  var idCol = -1, vendedorCol = -1;
  for (var c = 0; c < data[0].length; c++) {
    var h = String(data[0][c]).trim();
    if (h === 'IdOportunidade') idCol = c;
    if (h === 'IdVendedor') vendedorCol = c;
  }
  if (idCol === -1) return { ok: false, erro: 'Coluna IdOportunidade não encontrada.' };
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][idCol]) === String(idOportunidade)) {

      if (!ehDono && !isSolicitanteAdmin(body.solicitanteId)) {
        return { ok: false, erro: 'Você só pode excluir os próprios leads.' };
      }
      sheet.deleteRow(i + 1);
      return { ok: true };
    }
  }
  return { ok: false, erro: 'Lead não encontrado.' };
}

/**
 * action: salvarCliente — body { idCliente?, nome, tipoPessoa, cpfCnpj?, telefone,
 *   email?, endereco?, statusCliente?, vendedorResponsavel? }
 * Cria (sem idCliente EXISTENTE) ou atualiza (com idCliente já existente na
 * planilha) um cliente com TODOS os campos da aba Clientes. Usado tanto pela
 * tela CRUD de Clientes quanto pelo cadastro rápido dentro do Funil.
 *
 * IMPORTANTE: o navegador SEMPRE manda um idCliente (gerado na hora, pra
 * deixar a tela otimista/instantânea) — inclusive quando é um cliente NOVO.
 * Por isso a decisão create-vs-update aqui é baseada em ACHAR ou NÃO a linha
 * com esse ID na planilha, nunca em "o campo veio preenchido ou não".
 */
function actionSalvarCliente(body) {
  var nome = String(body.nome || '').trim();
  var telefone = String(body.telefone || '').trim();
  var tipoPessoa = String(body.tipoPessoa || '').trim();
  if (!nome || !telefone) return { ok: false, erro: 'Nome e telefone são obrigatórios.' };
  if (!tipoPessoa) return { ok: false, erro: 'Tipo de pessoa (Física/Jurídica) é obrigatório.' };

  var sheet = getSheetSafe(CONFIG.ABA_CLIENTES);
  if (!sheet) return { ok: false, erro: 'Aba Clientes não encontrada.' };
  var headers = getHeaderRow(sheet);
  var colIdx = {}; headers.forEach(function (h, i) { colIdx[h] = i + 1; });
  if (!colIdx['IdCliente']) return { ok: false, erro: 'Coluna IdCliente não encontrada.' };

  var campos = {};
  var nomeCol = colIdx['Nome Razao Social'] || colIdx['Nome'];
  if (nomeCol) campos[nomeCol] = nome;
  if (colIdx['Tipo Pessoa']) campos[colIdx['Tipo Pessoa']] = tipoPessoa;
  if (colIdx['CPF ou CNPJ'] && body.cpfCnpj !== undefined) campos[colIdx['CPF ou CNPJ']] = body.cpfCnpj;
  if (colIdx['Telefone']) campos[colIdx['Telefone']] = telefone;
  if (colIdx['Email'] && body.email !== undefined) campos[colIdx['Email']] = body.email;
  if (colIdx['Endereco'] && body.endereco !== undefined) campos[colIdx['Endereco']] = body.endereco;
  if (colIdx['Status Cliente'] && body.statusCliente) campos[colIdx['Status Cliente']] = body.statusCliente;
  if (colIdx['Vendedor Responsavel'] && body.vendedorResponsavel !== undefined) campos[colIdx['Vendedor Responsavel']] = body.vendedorResponsavel;
  if (colIdx['CPFEquatorial'] && body.cpfEquatorial !== undefined) campos[colIdx['CPFEquatorial']] = body.cpfEquatorial;
  if (colIdx['DataNascimentoEquatorial'] && body.dataNascimentoEquatorial) campos[colIdx['DataNascimentoEquatorial']] = parseDataISO(body.dataNascimentoEquatorial);

  // Procura a linha pelo ID recebido — SEM decidir create/update ainda.
  var rowIndex = -1;
  var data = sheet.getDataRange().getValues(); // sempre lê tudo — precisa pra achar a linha E pra checar telefone duplicado
  if (body.idCliente) {
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][colIdx['IdCliente'] - 1]) === String(body.idCliente)) { rowIndex = i + 1; break; }
    }
  }

  // Só checa duplicidade em CRIAÇÃO de cliente novo (rowIndex===-1), e só
  // bloqueia se o endereço vier em branco — se a pessoa preencheu um
  // endereço, presume-se que ela já conferiu/sabe que é um contato
  // diferente. O front-end já faz essa mesma checagem antes de chamar essa
  // ação (pra dar feedback na hora); isso aqui é só uma segunda trava, caso
  // dois vendedores cadastrem quase ao mesmo tempo.
  if (rowIndex === -1 && !body.confirmarClienteDiferente) {
    var enderecoNovoVazio = !String(body.endereco || '').trim();
    if (enderecoNovoVazio && colIdx['Telefone']) {
      var alvo8 = paraUltimos8Digitos(telefone);
      if (alvo8) {
        for (var j = 1; j < data.length; j++) {
          if (paraUltimos8Digitos(data[j][colIdx['Telefone'] - 1]) === alvo8) {
            return { ok: false, erro: 'Já existe um contato cadastrado para esse número', duplicidadeTelefone: true };
          }
        }
      }
    }
  }

  if (rowIndex !== -1) {
    // Achou a linha — é edição de um cliente que já existe.
    Object.keys(campos).forEach(function (col) { sheet.getRange(rowIndex, Number(col)).setValue(campos[col]); });
    return { ok: true, idCliente: body.idCliente };
  }

  // Não achou (ou não veio ID nenhum) — cria uma linha nova, usando o ID que
  // o navegador já gerou (otimista); só gera um novo aqui se por algum
  // motivo não tiver vindo nenhum.
  var novoId = body.idCliente || generateShortId();
  var lastCol = sheet.getLastColumn();
  var novaLinha = new Array(lastCol).fill('');
  Object.keys(campos).forEach(function (col) { novaLinha[Number(col) - 1] = campos[col]; });
  novaLinha[colIdx['IdCliente'] - 1] = novoId;
  sheet.getRange(sheet.getLastRow() + 1, 1, 1, novaLinha.length).setValues([novaLinha]);
  return { ok: true, idCliente: novoId, nome: nome, telefone: telefone };
}

/**
 * Atualiza campos cadastrais de um cliente existente na aba Clientes,
 * localizando a linha pelo IdCliente (coluna A).
 * Só sobrescreve os campos que vierem definidos no body (nome/cpf/endereco),
 * preservando os demais valores da linha (Telefone, Email, Status, etc).
 */
function updateCliente(body) {
  var idCliente = body.idCliente;
  if (!idCliente) return { ok: false, erro: 'idCliente é obrigatório.' };

  var sheet = getSheetSafe(CONFIG.ABA_CLIENTES);
  if (!sheet) return { ok: false, erro: 'Aba Clientes não encontrada.' };

  var data = sheet.getDataRange().getValues();
  if (data.length === 0) return { ok: false, erro: 'Aba Clientes está vazia.' };

  var headers = data[0].map(function (h) { return String(h).trim(); });
  var colIdx  = {};
  headers.forEach(function (h, i) { colIdx[h] = i; });

  var idCol = colIdx['IdCliente'];
  if (idCol === undefined) return { ok: false, erro: 'Coluna IdCliente não encontrada na aba Clientes.' };

  var rowIdx = -1;
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][idCol]) === String(idCliente)) { rowIdx = i + 1; break; }
  }
  if (rowIdx === -1) return { ok: false, erro: 'Cliente não encontrado (IdCliente ' + idCliente + ').' };

  var vendCol = colIdx['Vendedor Responsavel'];
  var dono = vendCol !== undefined ? String(data[rowIdx - 1][vendCol]) : null;
  var ehDono = dono && String(body.solicitanteId) === dono;
  if (!ehDono && !isSolicitanteAdmin(body.solicitanteId)) {
    return { ok: false, erro: 'Você só pode editar os próprios clientes.' };
  }

  var nomeCol = colIdx['Nome Razao Social'] !== undefined ? colIdx['Nome Razao Social'] : colIdx['Nome'];
  var cpfCol  = colIdx['CPF ou CNPJ']       !== undefined ? colIdx['CPF ou CNPJ']       : colIdx['CPF'];
  var endCol  = colIdx['Endereco'];

  if (body.nome !== undefined && body.nome !== null && nomeCol !== undefined) {
    sheet.getRange(rowIdx, nomeCol + 1).setValue(body.nome);
  }
  if (body.cpf !== undefined && body.cpf !== null && cpfCol !== undefined) {
    sheet.getRange(rowIdx, cpfCol + 1).setValue(body.cpf);
  }
  if (body.endereco !== undefined && body.endereco !== null && endCol !== undefined) {
    sheet.getRange(rowIdx, endCol + 1).setValue(body.endereco);
  }

  return {
    ok: true, acao: 'atualizado', idCliente: idCliente,
    nome: body.nome, cpf: body.cpf, endereco: body.endereco
  };
}

function saveProposal(body) {

  var propostaId = body.id || generateShortId();
  if (body.id) {
    for (var i = 1; i < values.length; i++) {
      if (String(values[i][0]) === String(body.id)) {
        rowIdx = i + 1; break;
      }
    }
  }

  var baseUrl  = (body.baseUrl || 'https://pampfelp.github.io/relatorio-ponto/proposta.html').replace(/\/$/, '');
  var linkProposta = baseUrl + '?id=' + propostaId;

  var rowData = [
    propostaId,
    body.data         || formatDateBR(new Date()),
    body.numero       || '',
    body.idOportunidade || '',
    body.nomeCliente  || '',
    body.idVendedor   || '',
    body.nomeVendedor || '',
    parseFloat(body.totalServicos)      || 0,
    parseFloat(body.totalDeslocamento)  || 0,
    parseFloat(body.total)              || 0,
    JSON.stringify(body.blocos || []),
    body.observacoes  || '',
    body.status       || 'Rascunho',
    linkProposta
  ];

  if (rowIdx !== -1) {
    sheet.getRange(rowIdx, 1, 1, rowData.length).setValues([rowData]);
    return { ok: true, acao: 'atualizado', id: propostaId, link: linkProposta };

    sheet.getRange(lastRow, linkCol).setFormula('=HYPERLINK("' + linkProposta.replace(/"/g, '""') + '","Ver proposta")');

  }

// Colunas de autenticação que devem existir na aba Vendedores.
// São criadas automaticamente (no fim da planilha) na primeira vez que
// alguém tenta logar, pedir reset, etc — sem apagar nada que já existe.
var AUTH_COLUNAS = ['SenhaHash', 'SenhaSalt', 'ResetToken', 'ResetTokenExpira'];

// Funções (valores da coluna "Tipo") que têm acesso total ao sistema —
// veem todo mundo em Ponto/Vendas/Funil e podem administrar a aba Usuários.
// Mantenha esta lista em sincronia com SG_ADMIN_ROLES no index.html.
var ADMIN_ROLES = ['admin', 'administrador', 'ceo', 'gestor', 'gerente'];

/**
 * Confere se o IdVendedor informado corresponde a alguém com função administrativa.
 * Usado para proteger as ações de gerenciamento de usuários.
 */
function isSolicitanteAdmin(solicitanteId) {
  if (!solicitanteId) return false;
  var ctx = ensureVendedoresAuthColumns();
  var sheet = ctx.sheet, colIdx = ctx.colIdx;
  var idCol = colIdx['IdVendedor'], tipoCol = colIdx['Tipo'];
  if (!idCol || !tipoCol) return false;

  var values = sheet.getDataRange().getValues();
  for (var i = 1; i < values.length; i++) {
    if (String(values[i][idCol - 1]) === String(solicitanteId)) {
      var tipo = String(values[i][tipoCol - 1] || '').trim().toLowerCase();
      return ADMIN_ROLES.indexOf(tipo) !== -1;
    }
  }
  return false;
}

/**
 * Garante que a aba Vendedores tem as colunas de autenticação.
 * Retorna {sheet, colIdx} onde colIdx é um mapa NomeDaColuna -> número da coluna (1-based).
 */
function ensureVendedoresAuthColumns() {
  var sheet = getSheet(CONFIG.ABA_VENDEDORES);
  var lastCol = Math.max(sheet.getLastColumn(), 1);
  var headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(function (h) { return String(h).trim(); });

  var faltando = AUTH_COLUNAS.filter(function (h) { return headers.indexOf(h) === -1; });
  if (faltando.length) {
    sheet.getRange(1, lastCol + 1, 1, faltando.length).setValues([faltando]);
  }

  var novoLastCol = sheet.getLastColumn();
  headers = sheet.getRange(1, 1, 1, novoLastCol).getValues()[0].map(function (h) { return String(h).trim(); });
  var colIdx = {};
  headers.forEach(function (h, i) { if (h) colIdx[h] = i + 1; }); // 1-based

  return { sheet: sheet, colIdx: colIdx };
}

function sha256Hex(str) {
  var bytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, str, Utilities.Charset.UTF_8);
  return bytes.map(function (b) {
    b = (b < 0) ? b + 256 : b;
    return (b < 16 ? '0' : '') + b.toString(16);
  }).join('');
}

function gerarSalt() {
  return Utilities.getUuid().replace(/-/g, '');
}

function hashSenha(senha, salt) {
  return sha256Hex(AUTH_PEPPER + '|' + salt + '|' + senha);
}

function gerarTokenNumerico() {
  return String(Math.floor(100000 + Math.random() * 900000)); // 6 dígitos
}

/**
 * Localiza a linha do vendedor pelo e-mail (case-insensitive).
 * Retorna null se não encontrado, ou { rowIndex, values, colIdx, sheet }.
 */
function findVendedorRowByEmail(email) {
  var ctx = ensureVendedoresAuthColumns();
  var sheet = ctx.sheet, colIdx = ctx.colIdx;
  var emailCol = colIdx['Email'];
  if (!emailCol) return null;

  var values = sheet.getDataRange().getValues();
  var alvo = String(email || '').trim().toLowerCase();
  if (!alvo) return null;

  for (var i = 1; i < values.length; i++) {
    if (String(values[i][emailCol - 1] || '').trim().toLowerCase() === alvo) {
      return { rowIndex: i + 1, values: values[i], colIdx: colIdx, sheet: sheet };
    }
  }
  return null;
}

/**
 * action: login — body { email, senha }
 */
function actionLogin(body) {
  var email = body.email, senha = body.senha;
  if (!email || !senha) return { ok: false, erro: 'Informe e-mail e senha.' };

  var found = findVendedorRowByEmail(email);
  if (!found) return { ok: false, erro: 'E-mail não encontrado.' };

  var colIdx = found.colIdx, row = found.values;

  var statusCol = colIdx['Status'];
  if (statusCol && String(row[statusCol - 1] || '').trim().toLowerCase() === 'inativo') {
    return { ok: false, erro: 'Usuário inativo. Fale com o administrador.' };
  }

  var hashCol = colIdx['SenhaHash'], saltCol = colIdx['SenhaSalt'];
  var hashArmazenado = String(row[hashCol - 1] || '');
  var salt = String(row[saltCol - 1] || '');

  var usuario = {
    idVendedor: row[colIdx['IdVendedor'] - 1],
    nome:       row[colIdx['Nome'] - 1],
    email:      row[colIdx['Email'] - 1],
    tipo:       colIdx['Tipo'] ? row[colIdx['Tipo'] - 1] : ''
  };

  if (!hashArmazenado) {
    // Ainda não tem senha própria: só entra com a senha padrão, e é
    // obrigado a trocar por uma senha só dele antes de acessar o sistema.
    if (senha !== SENHA_PADRAO) {
      return {
        ok: false, erro: 'SEM_SENHA',
        mensagem: 'Você ainda não tem uma senha própria. Use a senha padrão informada pela empresa para o primeiro acesso.'
      };
    }
    return { ok: true, usuario: usuario, precisaTrocarSenha: true };
  }

  if (hashSenha(senha, salt) !== hashArmazenado) {
    return { ok: false, erro: 'Senha incorreta.' };
  }

  return { ok: true, usuario: usuario };
}

/**
 * action: esqueciSenha — body { email }
 * Gera um código de 6 dígitos, salva na planilha com validade de 30 min
 * e envia por e-mail via MailApp (usa a conta Google dona do Apps Script).
 * Sempre responde ok:true (não revela se o e-mail existe, por segurança).
 */
function actionEsqueciSenha(body) {
  var email = body.email;
  if (!email) return { ok: false, erro: 'Informe o e-mail.' };

  var found = findVendedorRowByEmail(email);
  if (!found) return { ok: true }; // não revela se o e-mail existe

  var colIdx = found.colIdx, sheet = found.sheet, rowIndex = found.rowIndex;
  var token  = gerarTokenNumerico();
  var expira = new Date(Date.now() + 30 * 60 * 1000); // 30 minutos

  sheet.getRange(rowIndex, colIdx['ResetToken']).setValue(token);
  sheet.getRange(rowIndex, colIdx['ResetTokenExpira']).setValue(expira);

  var nome = found.values[colIdx['Nome'] - 1] || '';

  try {
    MailApp.sendEmail({
      to: email,
      subject: 'Solar Green · Código para redefinir sua senha',
      body:
        'Olá ' + nome + ',\n\n' +
        'Recebemos um pedido para redefinir a sua senha no sistema Solar Green.\n' +
        'Seu código de verificação é: ' + token + '\n\n' +
        'Esse código expira em 30 minutos. Se você não solicitou isso, pode ignorar este e-mail.\n\n' +
        '— Solar Green Suporte'
    });
  } catch (err) {
    return { ok: false, erro: 'Não foi possível enviar o e-mail: ' + err.message };
  }

  return { ok: true };
}

/**
 * action: redefinirSenha — body { email, token, novaSenha }
 * Usado no fluxo "esqueci minha senha" (não exige senha atual).
 */
function actionRedefinirSenha(body) {
  var email = body.email, token = body.token, novaSenha = body.novaSenha;
  if (!email || !token || !novaSenha) return { ok: false, erro: 'Preencha todos os campos.' };
  if (String(novaSenha).length < 6) return { ok: false, erro: 'A senha deve ter pelo menos 6 caracteres.' };

  var found = findVendedorRowByEmail(email);
  if (!found) return { ok: false, erro: 'E-mail não encontrado.' };

  var colIdx = found.colIdx, sheet = found.sheet, rowIndex = found.rowIndex, row = found.values;

  var tokenArmazenado = String(row[colIdx['ResetToken'] - 1] || '');
  var expiraVal = row[colIdx['ResetTokenExpira'] - 1];

  if (!tokenArmazenado || tokenArmazenado !== String(token).trim()) {
    return { ok: false, erro: 'Código inválido.' };
  }

  var expiraDate = (expiraVal instanceof Date) ? expiraVal : new Date(expiraVal);
  if (!expiraDate || isNaN(expiraDate.getTime()) || expiraDate.getTime() < Date.now()) {
    return { ok: false, erro: 'Código expirado. Solicite um novo.' };
  }

  var novoSalt = gerarSalt();
  var novoHash = hashSenha(novaSenha, novoSalt);

  sheet.getRange(rowIndex, colIdx['SenhaHash']).setValue(novoHash);
  sheet.getRange(rowIndex, colIdx['SenhaSalt']).setValue(novoSalt);
  sheet.getRange(rowIndex, colIdx['ResetToken']).setValue('');
  sheet.getRange(rowIndex, colIdx['ResetTokenExpira']).setValue('');

  return { ok: true };
}

/**
 * action: trocarSenha — body { idVendedor, senhaAtual, novaSenha }
 * Usado quando o usuário já está logado e quer trocar a própria senha
 * (não precisa de e-mail nem de token).
 */
function actionTrocarSenha(body) {
  var idVendedor = body.idVendedor, senhaAtual = body.senhaAtual, novaSenha = body.novaSenha;
  if (!idVendedor || !senhaAtual || !novaSenha) return { ok: false, erro: 'Preencha todos os campos.' };
  if (String(novaSenha).length < 6) return { ok: false, erro: 'A nova senha deve ter pelo menos 6 caracteres.' };

  var ctx = ensureVendedoresAuthColumns();
  var sheet = ctx.sheet, colIdx = ctx.colIdx;
  var values = sheet.getDataRange().getValues();
  var idCol = colIdx['IdVendedor'];

  var rowIndex = -1, row = null;
  for (var i = 1; i < values.length; i++) {
    if (String(values[i][idCol - 1]) === String(idVendedor)) { rowIndex = i + 1; row = values[i]; break; }
  }
  if (rowIndex === -1) return { ok: false, erro: 'Usuário não encontrado.' };

  var hashArmazenado = String(row[colIdx['SenhaHash'] - 1] || '');
  var salt = String(row[colIdx['SenhaSalt'] - 1] || '');

  if (hashArmazenado && hashSenha(senhaAtual, salt) !== hashArmazenado) {
    return { ok: false, erro: 'Senha atual incorreta.' };
  }

  var novoSalt = gerarSalt();
  var novoHash = hashSenha(novaSenha, novoSalt);
  sheet.getRange(rowIndex, colIdx['SenhaHash']).setValue(novoHash);
  sheet.getRange(rowIndex, colIdx['SenhaSalt']).setValue(novoSalt);

  return { ok: true };
}

/**

  if (!isSolicitanteAdmin(body.solicitanteId)) {

  }

  var ctx = ensureVendedoresAuthColumns();
  var sheet = ctx.sheet, colIdx = ctx.colIdx;

  var headers = Object.keys(colIdx);

  var values = sheet.getDataRange().getValues();
  var lista = [];
  for (var i = 1; i < values.length; i++) {
    var row = values[i];
    if (row.every(function (c) { return c === '' || c === null; })) continue;

      if (camposOcultos[h]) return;

      if (isDateLike(val)) val = formatDateBR(val);
      obj[h] = val;
    });

  }

}

/**
 * action: salvarVendedor — body { solicitanteId, idVendedor?, nome, email, telefone?, tipo?, status? }
 * Se idVendedor vier preenchido, atualiza a linha existente; senão, cria um vendedor novo.
 * Nunca mexe nas colunas de senha — isso só acontece por login/trocarSenha/redefinirSenha.
 */
function actionSalvarVendedor(body) {
  if (!isSolicitanteAdmin(body.solicitanteId)) {
    return { ok: false, erro: 'Acesso restrito a administradores.' };
  }

  var nome = (body.nome || '').trim(), email = (body.email || '').trim();
  if (!nome || !email) return { ok: false, erro: 'Nome e e-mail são obrigatórios.' };

  var ctx = ensureVendedoresAuthColumns();
  var sheet = ctx.sheet, colIdx = ctx.colIdx;
  var idCol = colIdx['IdVendedor'];

  // Se for edição, confere se o novo e-mail já não pertence a OUTRO vendedor.
  var values = sheet.getDataRange().getValues();
  var emailCol = colIdx['Email'];
  for (var k = 1; k < values.length; k++) {
    var mesmoEmail = emailCol && String(values[k][emailCol - 1] || '').trim().toLowerCase() === email.toLowerCase();
    var linhaDiferente = !body.idVendedor || String(values[k][idCol - 1]) !== String(body.idVendedor);
    if (mesmoEmail && linhaDiferente) {
      return { ok: false, erro: 'Já existe outro usuário com esse e-mail.' };
    }
  }

  var rowIndex = -1;
  if (body.idVendedor) {
    for (var i = 1; i < values.length; i++) {
      if (String(values[i][idCol - 1]) === String(body.idVendedor)) { rowIndex = i + 1; break; }
    }
  }

  var campos = {
    'Nome': nome,
    'Email': email
  };
  if (body.telefone !== undefined) campos['Telefone'] = body.telefone;
  if (body.tipo     !== undefined) campos['Tipo']     = body.tipo;
  if (body.status   !== undefined) campos['Status']   = body.status;

  if (rowIndex === -1) {
    var novoId = body.idVendedor || generateShortId(); // usa o ID que o cliente já gerou (otimista) — só gera aqui se não veio
    var lastCol = sheet.getLastColumn();
    var novaLinha = new Array(lastCol).fill('');
    Object.keys(colIdx).forEach(function (h) {
      if (h === 'IdVendedor') novaLinha[colIdx[h] - 1] = novoId;
      else if (campos[h] !== undefined) novaLinha[colIdx[h] - 1] = campos[h];
    });
    sheet.getRange(sheet.getLastRow() + 1, 1, 1, novaLinha.length).setValues([novaLinha]);
    invalidarCache('sh_vendedores');
    return { ok: true, acao: 'criado', idVendedor: novoId };
  } else {
    Object.keys(campos).forEach(function (h) {
      if (!colIdx[h]) return;
      sheet.getRange(rowIndex, colIdx[h]).setValue(campos[h]);
    });
    invalidarCache('sh_vendedores');
    return { ok: true, acao: 'atualizado', idVendedor: body.idVendedor };
  }
}

/**
 * action: salvarPermissoes — body { solicitanteId, permissoes: [{tipo,tela,permitido}] }
 * Só administradores. Substitui TODA a aba pelo conjunto enviado — é uma
 * matriz pequena (cargos x telas), então regravar tudo de uma vez é mais
 * simples e seguro do que sincronizar linha por linha.
 *
 * Se não existir nenhuma linha pra um par cargo+tela, o padrão (aplicado no
 * front-end) é PERMITIR — assim, telas novas que a gente adicionar no
 * futuro continuam visíveis pra todo mundo até você decidir restringir,
 * em vez de sumirem sozinhas sem aviso.
 */
function actionSalvarPermissoes(body) {
  if (!isSolicitanteAdmin(body.solicitanteId)) {
    return { ok: false, erro: 'Acesso restrito a administradores.' };
  }
  var lista = body.permissoes || [];
  var sheet = getOrCreateSheetComCabecalho(CONFIG.ABA_PERMISSOES, ['Tipo', 'Tela', 'Permitido']);
  var lastRow = sheet.getLastRow();
  if (lastRow > 1) sheet.getRange(2, 1, lastRow - 1, 3).clearContent();
  if (lista.length) {
    var linhas = lista.map(function (p) { return [p.tipo, p.tela, !!p.permitido]; });
    sheet.getRange(2, 1, linhas.length, 3).setValues(linhas);
  }
  invalidarCache('sh_permissoes');
  return { ok: true };
}

// Pasta de dados do app no Drive — é onde o AppSheet realmente guarda as fotos
// (caminho completo: Meu Drive / appsheet / data / CRMResolve-360750226 / <pasta da tabela> / <arquivo>).
// Os caminhos salvos na planilha só têm a parte final (ex: "Agendamentos Respostas_Images/xxx.png"),
// relativa a essa pasta — por isso a busca direta pelo nome nem sempre achava o arquivo certo.
var APPSHEET_PASTA_APP = 'CRMResolve-360750226';

function encontrarPastaPorNome(nome, pastaBase) {
  var it = pastaBase ? pastaBase.getFoldersByName(nome) : DriveApp.getFoldersByName(nome);
  return it.hasNext() ? it.next() : null;
}

/**
 * As fotos respondidas pelo AppSheet (antes do sistema novo) foram salvas como
 * um CAMINHO relativo dentro do Drive (ex: "Agendamentos Respostas_Images/xxx.png"),
 * relativo à pasta de dados do próprio app — não uma URL. Essa função resolve isso:
 * procura o arquivo primeiro dentro da pasta do app, depois em qualquer pasta com
 * esse nome, e por último em todo o Drive; gera o link público e REGRAVA a célula
 * com a URL — assim isso só roda uma vez por foto antiga.
 */
function resolverEMigrarFotoLegada(sheet, colIdx, rowIndex, valorAtual) {
  if (!valorAtual) return valorAtual;
  var valor = String(valorAtual).trim();

  // Link antigo no formato "uc?id=..." — troca pro formato "thumbnail?id=..." que
  // carrega direto como imagem com mais confiabilidade (e regrava a célula).
  var mUc = valor.match(/drive\.google\.com\/uc\?id=([\w-]+)/);
  if (mUc) {
    try {
      var arquivoAntigo = DriveApp.getFileById(mUc[1]);
      arquivoAntigo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    } catch (e) {
      // se não conseguir nem achar o arquivo pelo ID, segue o baile — só troca o formato do link mesmo
    }
    var urlNova = 'https://drive.google.com/thumbnail?id=' + mUc[1] + '&sz=w2000';
    if (colIdx['RespostaFoto']) sheet.getRange(rowIndex, colIdx['RespostaFoto']).setValue(urlNova);
    return urlNova;
  }

  if (/^https?:\/\//.test(valor)) return valor; // já é uma URL válida (outro formato conhecido)

  try {
    var partes = valor.split('/');
    var nomeArquivo = partes.pop();
    var nomePasta = partes.length ? partes.join('/') : null; // ex: "Agendamentos Respostas_Images"

    var arquivo = null;

    // 1) Dentro da pasta de dados do app (onde o AppSheet realmente guarda as fotos)
    var pastaApp = encontrarPastaPorNome(APPSHEET_PASTA_APP);
    if (pastaApp) {
      var pastaAlvo = nomePasta ? encontrarPastaPorNome(nomePasta, pastaApp) : pastaApp;
      if (pastaAlvo) {
        var it1 = pastaAlvo.getFilesByName(nomeArquivo);
        if (it1.hasNext()) arquivo = it1.next();
      }
    }

    // 2) Qualquer pasta com esse nome, em qualquer lugar do Drive
    if (!arquivo && nomePasta) {
      var pasta2 = encontrarPastaPorNome(nomePasta);
      if (pasta2) {
        var it2 = pasta2.getFilesByName(nomeArquivo);
        if (it2.hasNext()) arquivo = it2.next();
      }
    }

    // 3) Último recurso: procura o arquivo pelo nome em todo o Drive
    if (!arquivo) {
      var it3 = DriveApp.getFilesByName(nomeArquivo);
      if (it3.hasNext()) arquivo = it3.next();
    }

    if (!arquivo) return valor; // não achou de jeito nenhum — mantém como estava

    try { arquivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch (e) {}
    var url = 'https://drive.google.com/thumbnail?id=' + arquivo.getId() + '&sz=w2000';
    if (colIdx['RespostaFoto']) sheet.getRange(rowIndex, colIdx['RespostaFoto']).setValue(url);
    return url;
  } catch (err) {
    return valor; // se der qualquer erro na busca, não trava a tela — só não migra essa foto agora
  }
}

/**
 * Lê a aba Agendamentos Respostas migrando de passagem qualquer foto antiga
 * (caminho relativo do AppSheet) para uma URL de verdade.
 */
function lerAgendamentosRespostasComMigracao(filtroIdsAgendamentos) {
  var LIMITE_RESOLUCOES_POR_CHAMADA = 40; // já é por página (10 agendamentos), não a aba inteira — pode ser mais generoso
  var sheet = getOrCreateAgendamentosRespostasSheet();
  var values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];
  var headers = values[0].map(function (h) { return String(h).trim(); });
  var colIdx = {};
  headers.forEach(function (h, i) { if (h) colIdx[h] = i + 1; });
  var idAgCol = colIdx['IdAgendamento'];

  var out = [];
  var resolucoesFeitas = 0;
  for (var i = 1; i < values.length; i++) {
    var row = values[i];
    if (row.every(function (c) { return c === '' || c === null; })) continue;
    if (filtroIdsAgendamentos && idAgCol && !filtroIdsAgendamentos[row[idAgCol - 1]]) continue;
    var obj = {};
    headers.forEach(function (h, j) { if (h) obj[h] = row[j]; });
    var fotoAtual = colIdx['RespostaFoto'] ? String(obj['RespostaFoto'] || '') : '';
    var precisaMigrar = fotoAtual && !/^https?:\/\//.test(fotoAtual);
    var linkAntigoUc = fotoAtual && /drive\.google\.com\/uc\?id=/.test(fotoAtual);
    if (precisaMigrar || linkAntigoUc) {
      // Link "uc?id=" é só troca de formato (rápido, não conta na cota); caminho
      // relativo antigo do AppSheet precisa procurar no Drive (mais lento, tem cota).
      if (linkAntigoUc || resolucoesFeitas < LIMITE_RESOLUCOES_POR_CHAMADA) {
        if (precisaMigrar) resolucoesFeitas++;
        obj['RespostaFoto'] = resolverEMigrarFotoLegada(sheet, colIdx, i + 1, obj['RespostaFoto']);
      }
      // senão: deixa o caminho antigo como está por agora — o front-end mostra o aviso de
      // "link não encontrado" e essa foto é resolvida numa próxima vez que a tela recarregar.
    }
    out.push(obj);
  }
  return out;
}

function parseDataISO(s) {
  var m = String(s || '').trim().match(/^(\d{4})-(\d{2})-(\d{2})/);
  if (!m) return null;
  return new Date(parseInt(m[1], 10), parseInt(m[2], 10) - 1, parseInt(m[3], 10));
}

function findAgendamentoRowById(id) {
  if (!id) return null;
  var sheet  = getOrCreateAgendamentosSheet();
  var colIdx = agendamentosColIdx(sheet);
  var idCol  = colIdx['IdAgendamento'];
  if (!idCol) return null;
  var values = sheet.getDataRange().getValues();
  for (var i = 1; i < values.length; i++) {
    if (String(values[i][idCol - 1]) === String(id)) {
      return { sheet: sheet, colIdx: colIdx, rowIndex: i + 1, values: values[i] };
    }
  }
  return null;
}

/**
 * Um solicitante pode mexer num agendamento se for administrador
 * OU se for o técnico responsável por aquele agendamento específico.
 */
function isTecnicoOuAdmin(solicitanteId, idAgendamento) {
  if (!solicitanteId) return false;
  if (isSolicitanteAdmin(solicitanteId)) return true;
  var found = findAgendamentoRowById(idAgendamento);
  if (!found) return false;
  var tecnicoCol = found.colIdx['TecnicoResponsavel'];
  if (!tecnicoCol) return false;
  return String(found.values[tecnicoCol - 1]) === String(solicitanteId);
}

function minutosDeHoraGS(hhmm) {
  var m = String(hhmm || '').match(/^(\d{1,2}):(\d{2})/);
  if (!m) return null;
  return parseInt(m[1], 10) * 60 + parseInt(m[2], 10);
}

function normalizarTextoGS(s) {
  return String(s || '').trim().toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '');
}

function formatarDataBRGS(iso) {
  var m = String(iso || '').match(/^(\d{4})-(\d{2})-(\d{2})/);
  return m ? (m[3] + '/' + m[2] + '/' + m[1]) : String(iso || '');
}

/**
 * action: salvarAgendamento — body { idAgendamento?, idCliente, idContrato?, idServico,
 *   dataInicio ('YYYY-MM-DD'), horaInicio, horaFim, tecnicoResponsavel, statusAgendamento?,
 *   observacaoComercial?, quantidadeModulos?, modeloModulos?, quantidadeInversores?, modeloInversores? }
 * Criado por vendedor/admin — não exige que o solicitante seja o técnico.
 *
 * Réplica no servidor das mesmas regras da tela (que por sua vez replicam as
 * regras que já existiam no AppSheet): cliente precisa ter endereço, hora fim
 * > hora início, e sem conflito de horário pro mesmo técnico no mesmo dia.
 * Isso garante que ninguém consiga burlar a validação chamando a API direto.
 */
function actionSalvarAgendamento(body) {
  var idCliente = body.idCliente, idServico = body.idServico,
      dataInicio = body.dataInicio, tecnico = body.tecnicoResponsavel;
  if (!idCliente || !idServico || !dataInicio || !tecnico) {
    return { ok: false, erro: 'Cliente, serviço, data de início e técnico responsável são obrigatórios.' };
  }
  if (!body.horaInicio || !body.horaFim) {
    return { ok: false, erro: 'Escolha o horário de início e fim.' };
  }

  // Regra 1: cliente precisa ter endereço cadastrado antes de agendar.
  var clientes = sheetToObjects(getSheet(CONFIG.ABA_CLIENTES));
  var clienteObj = clientes.filter(function (c) { return String(c.IdCliente) === String(idCliente); })[0];
  if (!clienteObj || !String((clienteObj.Endereco || clienteObj['Endereço']) || '').trim()) {
    return { ok: false, erro: '⚠️ Este cliente não possui endereço cadastrado. Atualize o cadastro do cliente antes de realizar o agendamento.' };
  }

  // Regra 2: hora fim tem que ser maior que hora início.
  var minIni = minutosDeHoraGS(body.horaInicio), minFim = minutosDeHoraGS(body.horaFim);
  if (minIni === null || minFim === null || minFim <= minIni) {
    return { ok: false, erro: 'O horário fim deve ser maior que o início.' };
  }

  // Regra 3: sem conflito de horário pro mesmo técnico no mesmo dia (cancelados não contam).
  // O check + a gravação ficam dentro de um LockService: sem isso, duas requisições
  // simultâneas podem ler a planilha ANTES de qualquer uma escrever, nenhuma vê o
  // agendamento da outra, e as duas passam no check — dois agendamentos conflitantes
  // seriam salvos com sucesso ao mesmo tempo (é exatamente esse buraco que o
  // comentário da função já dizia que não devia existir).
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
  } catch (e) {
    return { ok: false, erro: 'Sistema ocupado, tente novamente em alguns segundos.' };
  }
  try {
    var todosAgendamentos = sheetToObjects(getOrCreateAgendamentosSheet());
    var conflito = todosAgendamentos.filter(function (a) {
      if (body.idAgendamento && String(a.IdAgendamento) === String(body.idAgendamento)) return false;
      if (String(a.TecnicoResponsavel) !== String(tecnico)) return false;
      if (normalizarTextoGS(a['Status Agendamento']) === 'cancelado') return false;
      if (normalizeDateKey(a['Data Inicio']) !== dataInicio) return false;
      var aIni = minutosDeHoraGS(a['Hora inicio']), aFim = minutosDeHoraGS(a['Hora Fim']);
      if (aIni === null || aFim === null) return false;
      return aIni < minFim && aFim > minIni;
    })[0];
    if (conflito) {
      var vendedores = getSheetObjectsComCache(CONFIG.ABA_VENDEDORES, 'sh_vendedores');
      var tecnicoObj = vendedores.filter(function (v) { return String(v.IdVendedor) === String(tecnico); })[0];
      var nomeTecnico = tecnicoObj ? tecnicoObj.Nome : tecnico;
      return { ok: false, erro: 'Conflito! O técnico ' + nomeTecnico + ' já possui agendamento neste horário no dia ' + formatarDataBRGS(dataInicio) + '.' };
    }

    var sheet  = getOrCreateAgendamentosSheet();
    var colIdx = agendamentosColIdx(sheet);
    var idCol  = colIdx['IdAgendamento'];

    var rowIndex = -1;
    if (body.idAgendamento) {
      var values = sheet.getDataRange().getValues();
      for (var i = 1; i < values.length; i++) {
        if (String(values[i][idCol - 1]) === String(body.idAgendamento)) { rowIndex = i + 1; break; }
      }
    }

    var campos = {
      'IdCliente': idCliente,
      'IdContrato': body.idContrato || '',
      'IdServico': idServico,
      'Data Inicio': parseDataISO(dataInicio),
      'Hora inicio': body.horaInicio || '',
      'Hora Fim': body.horaFim || '',
      'TecnicoResponsavel': tecnico,
      'Status Agendamento': body.statusAgendamento || 'Agendado',
      'Observacao Comercial': body.observacaoComercial || '',
      'Quantidade de Modulos': body.quantidadeModulos !== undefined ? body.quantidadeModulos : '',
      'Modelo Modulos': body.modeloModulos || '',
      'Quantidade Inversores': body.quantidadeInversores !== undefined ? body.quantidadeInversores : '',
      'Modelo Inversores': body.modeloInversores || ''
    };

    if (rowIndex === -1) {
      var novoId = body.idAgendamento || generateShortId(); // usa o ID que o cliente já gerou (otimista) — só gera aqui se não veio
      var lastCol = sheet.getLastColumn();
      var novaLinha = new Array(lastCol).fill('');
      Object.keys(colIdx).forEach(function (h) {
        if (h === 'IdAgendamento') novaLinha[colIdx[h] - 1] = novoId;
        else if (campos[h] !== undefined) novaLinha[colIdx[h] - 1] = campos[h];
      });
      sheet.getRange(sheet.getLastRow() + 1, 1, 1, novaLinha.length).setValues([novaLinha]);
      return { ok: true, acao: 'criado', idAgendamento: novoId };
    } else {
      Object.keys(campos).forEach(function (h) {
        if (!colIdx[h]) return;
        sheet.getRange(rowIndex, colIdx[h]).setValue(campos[h]);
      });
      return { ok: true, acao: 'atualizado', idAgendamento: body.idAgendamento };
    }
  } finally {
    lock.releaseLock();
  }
}

/**

function actionAtualizarStatusAgendamento(body) {

/**
 * Cria ou atualiza a resposta de um campo do template pra um agendamento
 * (upsert por IdAgendamento + IdTemplate). "campo" é qual coluna recebe o
 * valor: 'RespostaTexto', 'RespostaFoto' ou 'RespostaQuantidade' — são as
 * 3 colunas reais da aba (a mesma linha de resposta usa só uma delas).
 */
function salvarOuAtualizarResposta(idAgendamento, idTemplate, campo, valor) {
  var sheet  = getOrCreateAgendamentosRespostasSheet();
  var values = sheet.getDataRange().getValues();
  var headers = values.length ? values[0].map(function (h) { return String(h).trim(); }) : AGENDAMENTOS_RESPOSTAS_HEADERS.slice();
  var colIdx = {};
  headers.forEach(function (h, i) { colIdx[h] = i + 1; });
  if (!colIdx[campo]) return; // coluna não existe nessa planilha — não faz nada, evita erro

  var idAgCol = colIdx['IdAgendamento'], idTplCol = colIdx['IdTemplate'];
  for (var i = 1; i < values.length; i++) {
    if (String(values[i][idAgCol - 1]) === String(idAgendamento) &&
        String(values[i][idTplCol - 1]) === String(idTemplate)) {
      sheet.getRange(i + 1, colIdx[campo]).setValue(valor);
      return;
    }
  }

  var novaLinha = new Array(headers.length).fill('');
  if (colIdx['IdResposta']) novaLinha[colIdx['IdResposta'] - 1] = generateShortId();
  novaLinha[idAgCol - 1]    = idAgendamento;
  novaLinha[idTplCol - 1]   = idTemplate;
  novaLinha[colIdx[campo] - 1] = valor;
  sheet.getRange(sheet.getLastRow() + 1, 1, 1, novaLinha.length).setValues([novaLinha]);
}

/**
 * action: salvarRespostasAgendamento — body { solicitanteId, idAgendamento, respostas:[{idTemplate,resposta}] }
 * Salva em lote as respostas de texto/número/sim-ou-não/lista do formulário
 * (fotos vão por uploadFotoResposta, que já salva sozinho).
 */
function actionSalvarRespostasAgendamento(body) {
  var idAgendamento = body.idAgendamento, respostas = body.respostas || [];
  if (!idAgendamento) return { ok: false, erro: 'idAgendamento é obrigatório.' };
  if (!isTecnicoOuAdmin(body.solicitanteId, idAgendamento)) {
    return { ok: false, erro: 'Sem permissão para essa ordem de serviço.' };
  }

  var tplSheet = getSheetSafe(CONFIG.ABA_TEMPLATES);
  var templates = tplSheet ? sheetToObjects(tplSheet) : [];
  var tplById = {};
  templates.forEach(function (t) { if (t.IdTemplate) tplById[t.IdTemplate] = t; });

  respostas.forEach(function (r) {
    if (!r.idTemplate) return;
    var tpl = tplById[r.idTemplate];
    var tipo = tpl ? String(tpl.TipoInput || '').trim() : 'Texto';
    var campo = (tipo === 'Number') ? 'RespostaQuantidade' : 'RespostaTexto';
    var valor = (r.resposta === undefined || r.resposta === null) ? '' : r.resposta;
    salvarOuAtualizarResposta(idAgendamento, r.idTemplate, campo, valor);
  });

  var novoStatus = atualizarStatusAutomatico(idAgendamento, templates);
  return { ok: true, novoStatus: novoStatus };
}

/**
 * Roda depois de qualquer resposta salva (texto/número/lista OU foto).
 * Regra: se ainda está "Agendado" e já tem pelo menos uma resposta -> "Em Andamento".
 * Se TODAS as perguntas obrigatórias do serviço atual já têm resposta -> "Concluído".
 * Nunca mexe em agendamento já Cancelado ou já Concluído (decisão manual é respeitada),
 * e nunca trava o formulário — só atualiza o status na planilha.
 */
/**
 * A coluna "Obrigatorio" da aba Templates é uma caixa de seleção (checkbox)
 * do Sheets — o Apps Script lê isso como BOOLEANO true/false, não como o
 * texto "VERDADEIRO"/"FALSO" que aparece na planilha. Essa função reconhece
 * os dois formatos.
 */
function isObrigatorioGS(valor) {
  if (valor === true) return true;
  if (valor === false || valor === undefined || valor === null || valor === '') return false;
  var s = String(valor).trim().toUpperCase();
  return s === 'VERDADEIRO' || s === 'TRUE' || s === 'SIM' || s === '1';
}

function atualizarStatusAutomatico(idAgendamento, templatesCache) {
  var found = findAgendamentoRowById(idAgendamento);
  if (!found) return null;

  var statusCol = found.colIdx['Status Agendamento'];
  var statusAtual = statusCol ? String(found.values[statusCol - 1] || '').trim() : '';
  var statusNorm = statusAtual.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '');
  if (statusNorm === 'cancelado' || statusNorm === 'concluido') return null; // não mexe

  var idServico = found.values[found.colIdx['IdServico'] - 1];
  var templates = templatesCache;
  if (!templates) {
    var tplSheet = getSheetSafe(CONFIG.ABA_TEMPLATES);
    templates = tplSheet ? sheetToObjects(tplSheet) : [];
  }
  var obrigatorios = templates.filter(function (t) {
    return String(t.IdServico) === String(idServico) && isObrigatorioGS(t.Obrigatorio);
  });

  var todasRespostas = sheetToObjects(getOrCreateAgendamentosRespostasSheet())
    .filter(function (r) { return String(r.IdAgendamento) === String(idAgendamento); });
  var respostaPorTemplate = {};
  todasRespostas.forEach(function (r) { respostaPorTemplate[r.IdTemplate] = r; });

  function temValor(r) {
    return !!(r && (String(r.RespostaTexto || '').trim() || String(r.RespostaFoto || '').trim() || String(r.RespostaQuantidade || '').trim()));
  }

  var todosObrigatoriosRespondidos = obrigatorios.length > 0 && obrigatorios.every(function (t) {
    return temValor(respostaPorTemplate[t.IdTemplate]);
  });
  var temAlgumaResposta = todasRespostas.some(temValor);

  var novoStatus = null;
  if (todosObrigatoriosRespondidos) {
    novoStatus = 'Concluído';
  } else if (temAlgumaResposta && statusNorm === 'agendado') {
    novoStatus = 'Em Andamento';
  }

  if (novoStatus && novoStatus !== statusAtual) {
    found.sheet.getRange(found.rowIndex, statusCol).setValue(novoStatus);
    var statusNovoNorm = novoStatus.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '');
    if (statusNovoNorm === 'concluido' && found.colIdx['DataRelatorio']) {
      found.sheet.getRange(found.rowIndex, found.colIdx['DataRelatorio']).setValue(new Date());
    }
    return novoStatus;
  }
  return null;
}

/**
 * action: registrarPonto — body { solicitanteId, tipo, base64?, mimeType?, localizacao?, endereco? }
 * Usado pela tela de autoatendimento (Pontoeletronico.html) — o próprio
 * funcionário bate o ponto por ali. Tipo: "Entrada" | "Almoço" | "Retorno
 * Almoço" | "Saída". A selfie é opcional (algum funcionário pode não ter
 * câmera disponível), mas quando vem, sobe pro Drive igual às fotos de
 * agendamento.
 */
function actionRegistrarPonto(body) {
  var funcionario = body.solicitanteId, tipo = body.tipo;
  if (!funcionario || !tipo) return { ok: false, erro: 'Funcionário e tipo são obrigatórios.' };

  var vendedores = getSheetObjectsComCache(CONFIG.ABA_VENDEDORES, 'sh_vendedores');
  var pessoa = vendedores.filter(function (v) { return String(v.IdVendedor) === String(funcionario); })[0];
  if (!pessoa) return { ok: false, erro: 'Funcionário não encontrado.' };

  var fotoUrl = '';
  if (body.base64) {
    try {
      var bytes = Utilities.base64Decode(body.base64);
      var blob = Utilities.newBlob(bytes, body.mimeType || 'image/jpeg', (body.nomeArquivo || 'selfie') + '.jpg');
      var pasta = getOrCreatePastaFotosPonto();
      var arquivo = pasta.createFile(blob);
      arquivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      fotoUrl = 'https://drive.google.com/thumbnail?id=' + arquivo.getId() + '&sz=w2000';
    } catch (e) {
      // se a foto falhar, ainda assim registra o ponto sem foto — melhor
      // bater o ponto sem selfie do que não bater ponto nenhum.
    }
  }

  var sheet = getOrCreateSheetComCabecalho(CONFIG.ABA_PONTO,
    ['IdPonto', 'Funcionario', 'Tipo', 'Data e Hora', 'Foto Selfie', 'Localizacao', 'Endereço']);
  var headers = getHeaderRow(sheet);
  var colIdx = {}; headers.forEach(function (h, i) { colIdx[h] = i + 1; });

  var novoId = body.idPonto || generateShortId(); // usa o ID que o cliente já gerou (otimista) — só gera aqui se não veio
  var lastCol = sheet.getLastColumn();
  var novaLinha = new Array(lastCol).fill('');
  var campos = {
    'IdPonto': novoId, 'Funcionario': funcionario, 'Tipo': tipo,
    'Data e Hora': new Date(), 'Foto Selfie': fotoUrl,
    'Localizacao': body.localizacao || '', 'Endereço': body.endereco || ''
  };
  Object.keys(colIdx).forEach(function (h) { if (campos[h] !== undefined) novaLinha[colIdx[h] - 1] = campos[h]; });
  sheet.getRange(sheet.getLastRow() + 1, 1, 1, novaLinha.length).setValues([novaLinha]);

  return { ok: true, idPonto: novoId, nome: pessoa.Nome || '', fotoUrl: fotoUrl };
}

/**

 * Salva a foto no Google Drive (pasta "Solar Green - Fotos de Agendamentos"),

function actionUploadFotoResposta(body) {

    CONFIG.ABA_SERVICOS, CONFIG.ABA_FUNIL_LOG, CONFIG.ABA_FUNIL_SLA,
    CONFIG.ABA_PROPOSTAS, CONFIG.ABA_AGENDAMENTOS, CONFIG.ABA_AGENDAMENTOS_RESPOSTAS,
    CONFIG.ABA_TEMPLATES
  ];
  var resultado = esperadas.map(function (nome) {
    var achada = getSheetSafe(nome);
    return { esperada: nome, encontrada: achada ? achada.getName() : null, ok: !!achada };
  });
  return { ok: true, todasAsAbasDaPlanilha: todasAbas, checagem: resultado };
}
